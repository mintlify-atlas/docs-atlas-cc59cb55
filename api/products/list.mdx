---
title: "List Products"
api: "GET /products"
description: "Retrieve a paginated list of all active products"
---

## Overview

This endpoint returns a paginated list of all active products in the system. Only products with `active=true` are included in the response.

## Query Parameters

<ParamField query="page" type="integer" default="0">
  Page number (zero-based)
</ParamField>

<ParamField query="size" type="integer" default="20">
  Number of items per page
</ParamField>

<ParamField query="sort" type="string">
  Sort field and direction (e.g., `price,asc` or `title,desc`)
</ParamField>

## Response

Returns a paginated response containing an array of ProductDTO objects.

<ResponseField name="content" type="array">
  Array of product objects
  
  <ResponseField name="id" type="long">
    Product unique identifier
  </ResponseField>
  
  <ResponseField name="title" type="string">
    Product title/name
  </ResponseField>
  
  <ResponseField name="description" type="string">
    Product description
  </ResponseField>
  
  <ResponseField name="price" type="float">
    Product price
  </ResponseField>
  
  <ResponseField name="priceUnit" type="float">
    Unit price
  </ResponseField>
  
  <ResponseField name="stock" type="integer">
    Available stock quantity
  </ResponseField>
  
  <ResponseField name="mediaSrc" type="array">
    Array of image URLs
  </ResponseField>
  
  <ResponseField name="productCode" type="integer">
    Unique product code
  </ResponseField>
  
  <ResponseField name="calification" type="float">
    Average rating (0-5)
  </ResponseField>
  
  <ResponseField name="discount" type="float">
    Discount percentage
  </ResponseField>
  
  <ResponseField name="brand" type="object">
    Brand information
    
    <ResponseField name="id" type="long">
      Brand ID
    </ResponseField>
    
    <ResponseField name="name" type="string">
      Brand name
    </ResponseField>
    
    <ResponseField name="active" type="boolean">
      Brand active status
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="categories" type="array">
    Array of category objects
    
    <ResponseField name="id" type="long">
      Category ID
    </ResponseField>
    
    <ResponseField name="name" type="string">
      Category name
    </ResponseField>
    
    <ResponseField name="active" type="boolean">
      Category active status
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="isNew" type="boolean">
    Whether product is marked as new
  </ResponseField>
  
  <ResponseField name="isBestseller" type="boolean">
    Whether product is a bestseller
  </ResponseField>
  
  <ResponseField name="isFeatured" type="boolean">
    Whether product is featured
  </ResponseField>
  
  <ResponseField name="hero" type="boolean">
    Whether product is a hero product
  </ResponseField>
  
  <ResponseField name="active" type="boolean">
    Product active status
  </ResponseField>
</ResponseField>

<ResponseField name="pageable" type="object">
  Pagination information
  
  <ResponseField name="pageNumber" type="integer">
    Current page number
  </ResponseField>
  
  <ResponseField name="pageSize" type="integer">
    Items per page
  </ResponseField>
</ResponseField>

<ResponseField name="totalElements" type="long">
  Total number of products
</ResponseField>

<ResponseField name="totalPages" type="integer">
  Total number of pages
</ResponseField>

<ResponseField name="last" type="boolean">
  Whether this is the last page
</ResponseField>

## Example Request

<CodeGroup>

```bash cURL
curl -X GET "https://api.example.com/products?page=0&size=20" \
  -H "Content-Type: application/json"
```

```javascript JavaScript
const response = await fetch('https://api.example.com/products?page=0&size=20', {
  method: 'GET',
  headers: {
    'Content-Type': 'application/json'
  }
});
const data = await response.json();
```

```python Python
import requests

response = requests.get(
    'https://api.example.com/products',
    params={'page': 0, 'size': 20}
)
products = response.json()
```

</CodeGroup>

## Notes

- Only products with `active=true` are returned
- Default page size is 20 items
- Page numbers are zero-based (first page is 0)
- Results can be sorted using Spring Data's sort parameter format
